00061 // @fn          radio_reset
00062 // @brief       Reset radio core. 
00063 // @param       none
00064 // @return      none
00065 // *************************************************************************************************
00066 void radio_reset(void)
00067 {
00068         volatile u16 i;
00069         u8 x;
00070         
00071         // Reset radio core
00072         Strobe(RF_SRES);
00073         // Wait before checking IDLE 
00074         for (i=0; i<100; i++);
00075         do {
00076                 x = Strobe(RF_SIDLE);
00077         } while ((x&0x70)!=0x00);  
00078         
00079         // Clear radio error register
00080         RF1AIFERR = 0;
00081 }
