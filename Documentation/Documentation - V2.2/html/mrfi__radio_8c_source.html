00386    *    Radio power-up reset
00387    *   ----------------------
00388    */

00392   mrfiRadioInterfaceInit();

00394   /* Strobe Reset: Resets the radio and puts it in SLEEP state. */

00396 
00397   /* verify the correct radio is installed */
00398   MRFI_ASSERT( MRFI_RADIO_REG_READ( PARTNUM ) == MRFI_RADIO_PARTNUM );  /* incorrect radio specified */
00399   MRFI_ASSERT( MRFI_RADIO_REG_READ( VERSION ) == MRFI_RADIO_VERSION );  /* incorrect radio specified  */
00400 
00401   /* Put radio in Idle state */
00402   MRFI_STROBE_IDLE_AND_WAIT();
00403 
00404 
00405   /* ------------------------------------------------------------------
00406    *    Configure radio
00407    *   -----------------
00408    */
00409 
00410   /* Configure Radio interrupts:
00411    *
00412    * RF1AIN_0 => Programmed to PA_PD signal.
00413    *             Configure it to interrupt on falling edge.
00414    *
00415    * RF1AIN_1 => Programmed to RSSI Valid signal.
00416    *             No need to configure for interrupt. This value will be read
00417    *             through polling.
00418    *
00419    * RF1AIN_9 => Rising edge indicates SYNC sent/received and
00420    *             Falling edge indicates end of packet.
00421    *             Configure it to interrupt on falling edge.
00422    */
00423 
00424   /* Select Interrupt edge for PA_PD and SYNC signal:
00425    * Interrupt Edge select register: 1 == Interrupt on High to Low transition.
00426    */
00427   RF1AIES = BV(0) | BV(9);
00428 
00429   /* Write the power output to the PA_TABLE and verify the write operation.  */
00430   {
00431     uint8_t      readbackPATableValue = 0;
00432     bspIState_t  s;
00433 
00434     BSP_ENTER_CRITICAL_SECTION(s);
00435 
00436     while( !(RF1AIFCTL1 & RFINSTRIFG));
00437     RF1AINSTRW = 0x7E51;                    /* PA Table write (burst) */
00438 
00439     while( !(RF1AIFCTL1 & RFINSTRIFG));
00440     RF1AINSTRB = RF_SNOP;                   /* reset pointer */
00441 
00442     while( !(RF1AIFCTL1 & RFINSTRIFG));
00443     RF1AINSTRB = 0xFE;                      /* PA Table read (burst) */
00444 
00445     while( !(RF1AIFCTL1 & RFDINIFG));
00446     RF1ADINB    = 0x00;                     /* dummy write */
00447 
00448     while( !(RF1AIFCTL1 & RFDOUTIFG));
00449     readbackPATableValue = RF1ADOUT0B;
00450 
00451     MRFI_ASSERT(readbackPATableValue == 0x51);
00452 
00453     while( !(RF1AIFCTL1 & RFINSTRIFG));
00454     RF1AINSTRB = RF_SNOP;
00455 
00456     BSP_EXIT_CRITICAL_SECTION(s);
00457   }

00460   {

00470    */

00477     }

00491    * We will use the RSSI value to generate our random seed.
00492    */
00493

00495   MRFI_STROBE( SRX );
00496 
00497   /* delay for the rssi to be valid */
00498   MRFI_RSSI_VALID_WAIT();
00499 
00500   {
00501     uint8_t i;
00502     for(i=0; i<16; i++)
00503     {
00504       /* use most random bit of rssi to populate the random seed */
00505       mrfiRndSeed = (mrfiRndSeed << 1) | (MRFI_RADIO_REG_READ(RSSI) & 0x01);
00506     }
00507   }

00517    * using the SPWD cmd. However, it does wakes up if SXOFF was used to
00518    * put it to sleep.
00519    */

00525   /*****************************************************************************************
00526    *                            Compute reply delay scalar
00527    *
00528    * Formula from data sheet for all the narrow band radios is:
00529    *
00530    *                (256 + DATAR_Mantissa) * 2^(DATAR_Exponent)
00531    * DATA_RATE =    ------------------------------------------ * f(xosc)
00532    *                                    2^28
00533    *
00534    * To try and keep some accuracy we change the exponent of the denominator
00535    * to (28 - (exponent from the configuration register)) so we do a division
00536    * by a smaller number. We find the power of 2 by shifting.
00537    *
00538    * The maximum delay needed depends on the MAX_APP_PAYLOAD parameter. Figure
00539    * out how many bits that will be when overhead is included. Bits/bits-per-second
00540    * is seconds to transmit (or receive) the maximum frame. We multiply this number
00541    * by 1000 to find the time in milliseconds. We then additionally multiply by
00542    * 10 so we can add 5 and divide by 10 later, thus rounding up to the number of
00543    * milliseconds. This last won't matter for slow transmissions but for faster ones
00544    * we want to err on the side of being conservative and making sure the radio is on
00545    * to receive the reply. The semaphore monitor will shut it down. The delay adds in
00546    * a fudge factor that includes processing time on peer plus lags in Rx and processing
00547    * time on receiver's side.
00548    *
00549    * Note that we assume a 26 MHz clock for the radio...
00550    * ***************************************************************************************
00551    */
00552 #define   MRFI_RADIO_OSC_FREQ         26000000
00553 #define   PHY_PREAMBLE_SYNC_BYTES     8
00554 
00555   {
00556     uint32_t dataRate, bits;
00557     uint16_t exponent, mantissa;
00558 
00559     /* mantissa is in MDMCFG3 */
00560     mantissa = 256 + SMARTRF_SETTING_MDMCFG3;
00561 
00562     /* exponent is lower nibble of MDMCFG4. */
00563     exponent = 28 - (SMARTRF_SETTING_MDMCFG4 & 0x0F);
00564 
00565     /* we can now get data rate */
00566     dataRate = mantissa * (MRFI_RADIO_OSC_FREQ>>exponent);
00567 
00568     bits = ((uint32_t)((PHY_PREAMBLE_SYNC_BYTES + MRFI_MAX_FRAME_SIZE)*8))*10000;
00569 
00570     /* processing on the peer + the Tx/Rx time plus more */
00571     sReplyDelayScalar = PLATFORM_FACTOR_CONSTANT + (((bits/dataRate)+5)/10);
00572 
00573     /* This helper value is used to scale the backoffs during CCA. At very
00574      * low data rates we need to backoff longer to prevent continual sampling
00575      * of valid frames which take longer to send at lower rates. Use the scalar
00576      * we just calculated divided by 32. With the backoff algorithm backing
00577      * off up to 16 periods this will result in waiting up to about 1/2 the total
00578      * scalar value. For high data rates this does not contribute at all. Value
00579      * is in microseconds.
00580      */
00581     sBackoffHelper = MRFI_BACKOFF_PERIOD_USECS + (sReplyDelayScalar>>5)*1000;
00582   }
00583 
00584   /* Clean out buffer to protect against spurious frames */
00585   memset(mrfiIncomingPacket.frame, 0x00, sizeof(mrfiIncomingPacket.frame));
00586   memset(mrfiIncomingPacket.rxMetrics, 0x00, sizeof(mrfiIncomingPacket.rxMetrics));
00587 
00588   /* enable global interrupts */
00589   BSP_ENABLE_INTERRUPTS();
00590 }

00593 /**************************************************************************************************
00594  * @fn          MRFI_Transmit
00595  *
00596  * @brief       Transmit a packet using CCA algorithm.
00597  *
00598  * @param       pPacket - pointer to packet to transmit
00599  *
00600  * @return      Return code indicates success or failure of transmit:
00601  *                  MRFI_TX_RESULT_SUCCESS - transmit succeeded
00602  *                  MRFI_TX_RESULT_FAILED  - transmit failed because CCA failed
00603  **************************************************************************************************
00604  */
00605 uint8_t MRFI_Transmit(mrfiPacket_t * pPacket, uint8_t txType)
00606 {
00607   uint8_t ccaRetries;
00608   uint8_t txBufLen;
00609   uint8_t returnValue = MRFI_TX_RESULT_SUCCESS;
00610 
00611   /* radio must be awake to transmit */
00612   MRFI_ASSERT( mrfiRadioState != MRFI_RADIO_STATE_OFF );
00613 
00614   /* Turn off reciever. We can ignore/drop incoming packets during transmit. */
00615   Mrfi_RxModeOff();
00616 
00617   /* compute number of bytes to write to transmit FIFO */
00618   txBufLen = pPacket->frame[MRFI_LENGTH_FIELD_OFS] + MRFI_LENGTH_FIELD_SIZE;
00619 
00620   /* ------------------------------------------------------------------
00621    *    Write packet to transmit FIFO
00622    *   --------------------------------
00623    */
00624   MRFI_RADIO_WRITE_TX_FIFO(&(pPacket->frame[0]), txBufLen);
00625 
00626 
00627   /* ------------------------------------------------------------------
00628    *    Immediate transmit
00629    *   ---------------------
00630    */
00631   if (txType == MRFI_TX_TYPE_FORCED)
00632   {
00633     /* Issue the TX strobe. */
00634     MRFI_STROBE( STX );
00635 
00636     /* Wait for transmit to complete */
00637     while(!MRFI_SYNC_PIN_INT_FLAG_IS_SET());
00638 
00639     /* Clear the interrupt flag */
00640     MRFI_CLEAR_SYNC_PIN_INT_FLAG();
00641   }
00642   else
00643   {
00644     /* ------------------------------------------------------------------
00645      *    CCA transmit
00646      *   ---------------
00647      */
00648 
00649     MRFI_ASSERT( txType == MRFI_TX_TYPE_CCA );
00650 
00651     /* set number of CCA retries */
00652     ccaRetries = MRFI_CCA_RETRIES;
00653 
00654 
00655     /* ===============================================================================
00656      *    Main Loop
00657      *  =============
00658      */
00659     for (;;)
00660     {
00661       /* Radio must be in RX mode for CCA to happen.
00662        * Otherwise it will transmit without CCA happening.
00663        */
00664 
00665       /* Can not use the Mrfi_RxModeOn() function here since it turns on the
00666        * Rx interrupt, which we don't want in this case.
00667        */
00668       MRFI_STROBE( SRX );
00669 
00670       /* wait for the rssi to be valid. */
00671       MRFI_RSSI_VALID_WAIT();
00672 
00673       /*
00674        *  Clear the PA_PD pin interrupt flag.  This flag, not the interrupt itself,
00675        *  is used to capture the transition that indicates a transmit was started.
00676        *  The pin level cannot be used to indicate transmit success as timing may
00677        *  prevent the transition from being detected.  The interrupt latch captures
00678        *  the event regardless of timing.
00679        */
00680       MRFI_CLEAR_PAPD_PIN_INT_FLAG();
00681 
00682       /* send strobe to initiate transmit */
00683       MRFI_STROBE( STX );
00684 
00685       /* Delay long enough for the PA_PD signal to indicate a
00686        * successful transmit. This is the 250 XOSC periods
00687        * (9.6 us for a 26 MHz crystal).
00688        * Found out that we need a delay of atleast 25 us on CC1100 to see
00689        * the PA_PD signal change. Hence keeping the same for CC430
00690        */
00691       Mrfi_DelayUsec(25);
00692 
00693 
00694       /* PA_PD signal goes from HIGH to LOW when going from RX to TX state.
00695        * This transition is trapped as a falling edge interrupt flag
00696        * to indicate that CCA passed and the transmit has started.
00697        */
00698       if (MRFI_PAPD_INT_FLAG_IS_SET())
00699       {
00700         /* ------------------------------------------------------------------
00701         *    Clear Channel Assessment passed.
00702         *   ----------------------------------
00703         */
00704 
00705         /* Clear the PA_PD int flag */
00706         MRFI_CLEAR_PAPD_PIN_INT_FLAG();
00707 
00708         /* PA_PD signal stays LOW while in TX state and goes back to HIGH when
00709          * the radio transitions to RX state.
00710          */
00711         /* wait for transmit to complete */
00712         while (!MRFI_PAPD_PIN_IS_HIGH());
00713 
00714         /* transmit done, break */
00715         break;
00716       }
00717       else
00718       {
00719         /* ------------------------------------------------------------------
00720          *    Clear Channel Assessment failed.
00721          *   ----------------------------------
00722          */
00723 
00724         /* Turn off radio and save some power during backoff */
00725 
00726         /* NOTE: Can't use Mrfi_RxModeOff() - since it tries to update the
00727          * sync signal status which we are not using during the TX operation.
00728          */
00729         MRFI_STROBE_IDLE_AND_WAIT();
00730 
00731         /* flush the receive FIFO of any residual data */
00732         MRFI_STROBE( SFRX );
00733 
00734         /* Retry ? */
00735         if (ccaRetries != 0)
00736         {
00737           /* delay for a random number of backoffs */
00738           Mrfi_RandomBackoffDelay();
00739 
00740           /* decrement CCA retries before loop continues */
00741           ccaRetries--;
00742         }
00743         else /* No CCA retries are left, abort */
00744         {
00745           /* set return value for failed transmit and break */
00746           returnValue = MRFI_TX_RESULT_FAILED;
00747           break;
00748         }
00749       } /* CCA Failed */
00750     } /* CCA loop */
00751   }/* txType is CCA */
00752 
00753 
00754   /* Done with TX. Clean up time... */
00755 
00756   /* Radio is already in IDLE state */
00757 
00758   /*
00759    * Flush the transmit FIFO.  It must be flushed so that
00760    * the next transmit can start with a clean slate.
00761    */
00762   MRFI_STROBE( SFTX );
00763 
00764   /* If the radio was in RX state when transmit was attempted,
00765    * put it back to Rx On state.
00766    */
00767   if(mrfiRadioState == MRFI_RADIO_STATE_RX)
00768   {
00769     Mrfi_RxModeOn();
00770   }
00771 
00772   return( returnValue );
00773 }

00792 /**************************************************************************************************
00793  * @fn          Mrfi_SyncPinRxIsr
00794  *
00795  * @brief       This interrupt is called when the SYNC signal transition from high to low.
00796  *              The sync signal is routed to the sync pin which is a GPIO pin.  This high-to-low
00797  *              transition signifies a receive has completed.  The SYNC signal also goes from
00798  *              high to low when a transmit completes.   This is protected against within the
00799  *              transmit function by disabling sync pin interrupts until transmit completes.
00800  *
00801  * @param       none
00802  *
00803  * @return      none
00804  **************************************************************************************************
00805  */
00806 static void Mrfi_SyncPinRxIsr(void)
00807 {
00808   uint8_t frameLen = 0x00;
00809   uint8_t rxBytes;
00810 
00811   /* We should receive this interrupt only in RX state
00812    * Should never receive it if RX was turned On only for
00813    * some internal mrfi processing like - during CCA.
00814    * Otherwise something is terribly wrong.
00815    */
00816   MRFI_ASSERT( mrfiRadioState == MRFI_RADIO_STATE_RX );
00817 
00818   /* ------------------------------------------------------------------
00819    *    Get RXBYTES
00820    *   -------------
00821    */
00822 
00823   /*
00824    *  Read the RXBYTES register from the radio.
00825    *  Bit description of RXBYTES register:
00826    *    bit 7     - RXFIFO_OVERFLOW, set if receive overflow occurred
00827    *    bits 6:0  - NUM_BYTES, number of bytes in receive FIFO
00828    *
00829    *  Due a chip bug, the RXBYTES register must read the same value twice
00830    *  in a row to guarantee an accurate value.
00831    */
00832   {
00833     uint8_t rxBytesVerify;
00834 
00835     rxBytesVerify = MRFI_RADIO_REG_READ( RXBYTES );
00836 
00837     do
00838     {
00839       rxBytes = rxBytesVerify;
00840       rxBytesVerify = MRFI_RADIO_REG_READ( RXBYTES );
00841     }
00842     while (rxBytes != rxBytesVerify);
00843   }
00844 
00845 
00846   /* ------------------------------------------------------------------
00847    *    FIFO empty?
00848    *   -------------
00849    */
00850 
00851   /*
00852    *  See if the receive FIFIO is empty before attempting to read from it.
00853    *  It is possible nothing the FIFO is empty even though the interrupt fired.
00854    *  This can happen if address check is enabled and a non-matching packet is
00855    *  received.  In that case, the radio automatically removes the packet from
00856    *  the FIFO.
00857    */
00858   if (rxBytes == 0)
00859   {
00860     /* receive FIFO is empty - do nothing, skip to end */
00861   }
00862   else
00863   {
00864     /* receive FIFO is not empty, continue processing */
00865 
00866     /* ------------------------------------------------------------------
00867      *    Process frame length
00868      *   ----------------------
00869      */
00870 
00871     /* read the first byte from FIFO - the packet length */
00872     MRFI_RADIO_READ_RX_FIFO(&frameLen, MRFI_LENGTH_FIELD_SIZE);
00873 
00874 
00875     /*
00876      *  Make sure that the frame length just read corresponds to number of bytes in the buffer.
00877      *  If these do not match up something is wrong.
00878      *
00879      *  This can happen for several reasons:
00880      *   1) Incoming packet has an incorrect format or is corrupted.
00881      *   2) The receive FIFO overflowed.  Overflow is indicated by the high
00882      *      bit of rxBytes.  This guarantees the value of rxBytes value will not
00883      *      match the number of bytes in the FIFO for overflow condition.
00884      *   3) Interrupts were blocked for an abnormally long time which
00885      *      allowed a following packet to at least start filling the
00886      *      receive FIFO.  In this case, all received and partially received
00887      *      packets will be lost - the packet in the FIFO and the packet coming in.
00888      *      This is the price the user pays if they implement a giant
00889      *      critical section.
00890      *   4) A failed transmit forced radio to IDLE state to flush the transmit FIFO.
00891      *      This could cause an active receive to be cut short.
00892      *
00893      *  Also check the sanity of the length to guard against rogue frames.
00894      */
00895     if ((rxBytes != (frameLen + MRFI_LENGTH_FIELD_SIZE + MRFI_RX_METRICS_SIZE))           ||
00896         ((frameLen + MRFI_LENGTH_FIELD_SIZE) > MRFI_MAX_FRAME_SIZE) ||
00897         (frameLen < MRFI_MIN_SMPL_FRAME_SIZE)
00898        )
00899     {
00900       bspIState_t s;
00901       noFrame++;
00902 
00903       /* mismatch between bytes-in-FIFO and frame length */
00904 
00905       /*
00906        *  Flush receive FIFO to reset receive.  Must go to IDLE state to do this.
00907        *  The critical section guarantees a transmit does not occur while cleaning up.
00908        */
00909       BSP_ENTER_CRITICAL_SECTION(s);
00910       MRFI_STROBE_IDLE_AND_WAIT();
00911       MRFI_STROBE( SFRX );
00912       MRFI_STROBE( SRX );
00913       BSP_EXIT_CRITICAL_SECTION(s);
00914 
00915       /* flush complete, skip to end */
00916     }
00917     else
00918     {
00919       /* bytes-in-FIFO and frame length match up - continue processing */
00920 
00921       /* ------------------------------------------------------------------
00922        *    Get packet
00923        *   ------------
00924        */
00925 
00926       /* clean out buffer to help protect against spurious frames */
00927       memset(mrfiIncomingPacket.frame, 0x00, sizeof(mrfiIncomingPacket.frame));
00928 
00929       /* set length field */
00930       mrfiIncomingPacket.frame[MRFI_LENGTH_FIELD_OFS] = frameLen;
00931 
00932       /* get packet from FIFO */
00933       MRFI_RADIO_READ_RX_FIFO(&(mrfiIncomingPacket.frame[MRFI_FRAME_BODY_OFS]), frameLen);
00934 
00935       /* get receive metrics from FIFO */
00936       MRFI_RADIO_READ_RX_FIFO(&(mrfiIncomingPacket.rxMetrics[0]), MRFI_RX_METRICS_SIZE);
00937 
00938 
00939       /* ------------------------------------------------------------------
00940        *    CRC check
00941        *   ------------
00942        */
00943 
00944       /*
00945        *  Note!  Automatic CRC check is not, and must not, be enabled.  This feature
00946        *  flushes the *entire* receive FIFO when CRC fails.  If this feature is
00947        *  enabled it is possible to be reading from the FIFO and have a second
00948        *  receive occur that fails CRC and automatically flushes the receive FIFO.
00949        *  This could cause reads from an empty receive FIFO which puts the radio
00950        *  into an undefined state.
00951        */
00952 
00953       /* determine if CRC failed */
00954       if (!(mrfiIncomingPacket.rxMetrics[MRFI_RX_METRICS_CRC_LQI_OFS] & MRFI_RX_METRICS_CRC_OK_MASK))
00955       {
00956         /* CRC failed - do nothing, skip to end */
00957         crcFail++;
00958       }
00959       else
00960       {
00961         /* CRC passed - continue processing */
00962         crcPass++;
00963 
00964         /* ------------------------------------------------------------------
00965          *    Filtering
00966          *   -----------
00967          */
00968 
00969         /* if address is not filtered, receive is successful */
00970         if (!Mrfi_RxAddrIsFiltered(MRFI_P_DST_ADDR(&mrfiIncomingPacket)))
00971         {
00972           {
00973             /* ------------------------------------------------------------------
00974              *    Receive successful
00975              *   --------------------
00976              */
00977 
00978             /* Convert the raw RSSI value and do offset compensation for this radio */
00979             mrfiIncomingPacket.rxMetrics[MRFI_RX_METRICS_RSSI_OFS] =
00980                 Mrfi_CalculateRssi(mrfiIncomingPacket.rxMetrics[MRFI_RX_METRICS_RSSI_OFS]);
00981 
00982             /* Remove the CRC valid bit from the LQI byte */
00983             mrfiIncomingPacket.rxMetrics[MRFI_RX_METRICS_CRC_LQI_OFS] =
00984               (mrfiIncomingPacket.rxMetrics[MRFI_RX_METRICS_CRC_LQI_OFS] & MRFI_RX_METRICS_LQI_MASK);
00985 
00986 
00987             /* call external, higher level "receive complete" processing routine */
00988             MRFI_RxCompleteISR();
00989           }
00990         }
00991       }
00992     }
00993   }
00994 
00995   /* ------------------------------------------------------------------
00996    *    End of function
00997    *   -------------------
00998    */
00999 }

01163 /**************************************************************************************************

01170  *
01171  * @return      -

01173  */
01174 // [BM] Changed because CC1101_VECTOR ISR is declared of source code and this handler is called indirect
01175 //BSP_ISR_FUNCTION( MRFI_RadioIsr, CC1101_VECTOR )
01176 void MRFI_RadioIsr(void)
01177 {
01178   uint16_t coreIntSource = RF1AIV;            /* Radio Core      interrupt register */
01179   uint16_t interfaceIntSource = RF1AIFIV;     /* Radio Interface interrupt register */
01180 
01181   /* Interface interrupt */
01182   if(interfaceIntSource)
01183   {
01184     if(interfaceIntSource == RF1AIFIV_RFERRIFG)
01185     {
01186       uint16_t interfaceError = RF1AIFERRV;
01187 
01188       if(interfaceError == RF1AIFERRV_LVERR)
01189       {
01190         /* Low core voltage error */
01191       }
01192       else if(interfaceError == RF1AIFERRV_OPERR)
01193       {
01194         /* Operand error */
01195       }
01196       else if(interfaceError == RF1AIFERRV_OUTERR)
01197       {
01198         /* output data not available error */
01199       }
01200       else if(interfaceError == RF1AIFERRV_OPOVERR)
01201       {
01202         /* Operand overwrite error */
01203       }
01204       else
01205       {
01206         /* Can't possibly happen. If interface error flag was set,
01207          * then one of the interface errors must be set.
01208          */
01209         MRFI_FORCE_ASSERT();
01210       }
01211 
01212       /* Assert anyways. No error handling implemented. */
01213       MRFI_FORCE_ASSERT();
