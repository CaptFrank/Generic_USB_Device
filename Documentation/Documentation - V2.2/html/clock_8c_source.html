00113 // @brief       Check the Unified Clock System.
00114 // @param       none
00115 // @return      none
00116 // *************************************************************************************************
00117 void check_ucs(void) 
00118 {
00119     // Incremet counter every minute if started
00120     if(sUcsResetHandling.counter != 0) 
00121     { 
00122         sUcsResetHandling.counter++; 
00123         // Reset UCS twice: one times after one minute and a second time after an other minute
00124         if((sUcsResetHandling.counter == 2) || (sUcsResetHandling.counter == 3))
00125         { 
00126           // Reset UCS
00127           __disable_interrupt();
00128           init_ucs();
00129           __enable_interrupt();
00130         }
00131         // Switch off symbol on display after 12 hours (60*12=720 minutes) and stop counter
00132         if(sUcsResetHandling.counter >= 720) 
00133         { 
00134           sUcsResetHandling.counter = 0;
00135           display_symbol(LCD_SYMB_MAX, SEG_OFF);
00136         }
00137     }
00138         
00139     // Check UCS for failure
00140     if((SFRIFG1 & OFIFG) != 0)
00141     {
00142         display_symbol(LCD_SYMB_MAX, SEG_ON);
00143         if(sUcsResetHandling.status == UCSRESET_AUTO) 
00144         { 
00145           // Start counter
00146           sUcsResetHandling.counter = 1;
00147         }
00148     }
