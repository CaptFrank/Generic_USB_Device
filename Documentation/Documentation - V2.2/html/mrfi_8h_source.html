00066 /* Network header size definition */
00067 /* *********************************  NOTE  ****************************************
00068  * There is a dependency here that really shouldn't exist. A reimplementation
00069  * is necessary to avoid it.
00070  *
00071  * MRFI allocates the frame buffer which means it needs to know at compile time
00072  * how big the buffer is. This means in must know the NWK header size, the
00073  * maximum NWK and User application payload sizes and whether Security is enabled.
00074  * ********************************************************************************
00075  */
00076 #ifndef SMPL_SECURE
00077 #define  NWK_HDR_SIZE   3
00078 #define  NWK_PAYLOAD    MAX_NWK_PAYLOAD
00079 #else
00080 #define  NWK_HDR_SIZE   6
00081 #define  NWK_PAYLOAD    (MAX_NWK_PAYLOAD+4)
00082 #endif
00083 
00084 /* if external code has defined a maximum payload, use that instead of default */
00085 #ifdef MAX_APP_PAYLOAD
00086 #ifndef MAX_NWK_PAYLOAD
00087 #error ERROR: MAX_NWK_PAYLOAD not defined
00088 #endif
00089 #if MAX_APP_PAYLOAD < NWK_PAYLOAD
00090 #define MAX_PAYLOAD  NWK_PAYLOAD
00091 #else
00092 #define MAX_PAYLOAD  MAX_APP_PAYLOAD
00093 #endif
00094 #define MRFI_MAX_PAYLOAD_SIZE  (MAX_PAYLOAD+NWK_HDR_SIZE) /* SimpliciTI payload size plus six byte overhead */
00095 #endif
