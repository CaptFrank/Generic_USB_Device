00105  * @param       addr - address of register to strobe

00109  */

00111 {
00112   uint8_t statusByte, gdoState;

00114 
00115   /* Check for invalid address.
00116    * 0xBD is for SNOP with MSP set to read the bytes available in RX FIFO.
00117    */
00118   MRFI_RIF_ASSERT( (addr == 0xBD) || (addr >= RF_SRES) && (addr <= RF_SNOP));
00119 
00120   /* Lock out access to Radio IF */
00121   MRFI_RIF_ENTER_CRITICAL_SECTION(s);
00122 
00123   /* Clear the Status read flag */
00124   MRFI_RADIO_STATUS_READ_CLEAR();
00125 
00126   /* Wait for radio to be ready for next instruction */
00127   MRFI_RADIO_INST_WRITE_WAIT();
00128 
00129   if ((addr > RF_SRES) && (addr < RF_SNOP))
00130   {
00131     /* buffer IOCFG2 state */
00132     gdoState = MRFI_RADIO_REG_READ(IOCFG2);
00133 
00134     /* c-ready to GDO2 */
00135     MRFI_RADIO_REG_WRITE(IOCFG2, 0x29);
00136 
00137     RF1AINSTRB = addr;
00138 
00139     /* chip at sleep mode */
00140     if ((RF1AIN & 0x04) == 0x04)
00141     {
00142       if ( (addr == RF_SXOFF) || (addr == RF_SPWD) || (addr == RF_SWOR) )
00143       {
00144         /* Do nothing */
00145       }
00146       else
00147       {
00148         /* c-ready */
00149         while ((RF1AIN & 0x04) == 0x04);
00150 
00151         /* Delay should be 760us */
00152         Mrfi_DelayUsec(760);
00153       }
00154     }
00155 
00156     /* restore IOCFG2 setting */
00157     MRFI_RADIO_REG_WRITE(IOCFG2, gdoState);
00158   }
00159   else
00160   {
00161     /* chip active mode */
00162     RF1AINSTRB = addr;
00163   }
00164

00166   statusByte = RF1ASTAT0B;
00167 
00168   /* Allow access to Radio IF */
00169   MRFI_RIF_EXIT_CRITICAL_SECTION(s);
