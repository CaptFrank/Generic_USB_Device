00056  *         Any compiler worth its salt will optimize these conditionals out for the
00057  *         smallest possible code.
00058  */

00060 #define __bsp_LED_TURN_ON__(bit,port,ddr,low)  \
00061   st( if (low) { port &= ~BV(bit); } else { port |= BV(bit); } )
00062 
00063 #define __bsp_LED_TURN_OFF__(bit,port,ddr,low)  \
00064   st( if (low) { port |= BV(bit); } else { port &= ~BV(bit); } )
00065 
00066 #define __bsp_LED_IS_ON__(bit,port,ddr,low)  \
00067   ( (low) ? (!((port) & BV(bit))) : ((port) & BV(bit)) )
00068 
00069 #define __bsp_LED_TOGGLE__(bit,port,ddr,low)     st( port ^= BV(bit); )
00070 #define __bsp_LED_CONFIG__(bit,port,ddr,low)     st( ddr |= BV(bit); )
