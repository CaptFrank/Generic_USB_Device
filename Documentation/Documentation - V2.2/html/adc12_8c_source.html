00114 // *************************************************************************************************
00115 // @fn          adc12_single_conversion
00116 // @brief       Init ADC12. Do single conversion. Turn off ADC12.
00117 // @param       none
00118 // @return      none
00119 // *************************************************************************************************
00120 u16 adc12_single_conversion(u16 ref, u16 sht, u16 channel)
00121 {
00122         // Initialize the shared reference module 
00123         REFCTL0 |= REFMSTR + ref + REFON;               // Enable internal reference (1.5V or 2.5V)
00124   
00125         // Initialize ADC12_A 
00126         ADC12CTL0 = sht + ADC12ON;                                      // Set sample time 
00127         ADC12CTL1 = ADC12SHP;                           // Enable sample timer
00128         ADC12MCTL0 = ADC12SREF_1 + channel;             // ADC input channel  
00129         ADC12IE = 0x001;                                // ADC_IFG upon conv result-ADCMEMO
00130   
00131         // Wait 2 ticks (66us) to allow internal reference to settle
00132         Timer0_A4_Delay(2);       
00133         
00134         // Start ADC12
00135         ADC12CTL0 |= ADC12ENC;                                                  
00136 
00137         // Clear data ready flag
00138         adc12_data_ready = 0;
00139         
00140         // Sampling and conversion start  
00141     ADC12CTL0 |= ADC12SC;                       
00142     
00143     // Wait until ADC12 has finished
00144     Timer0_A4_Delay(5);       
00145         while (!adc12_data_ready);
00146         
00147         // Shut down ADC12
00148         ADC12CTL0 &= ~(ADC12ENC | ADC12SC | sht);
00149         ADC12CTL0 &= ~ADC12ON;
00150         
00151         // Shut down reference voltage  
00152         REFCTL0 &= ~(REFMSTR + ref + REFON); 
00153         
00154         ADC12IE = 0;                            
00155         
00156         // Return ADC result
00157         return (adc12_result);
00158 }

00170 __interrupt void ADC12ISR (void)
00171 {
00172   switch(__even_in_range(ADC12IV,34))
00173   {
00174   case  0: break;                           // Vector  0:  No interrupt
00175   case  2: break;                           // Vector  2:  ADC overflow
00176   case  4: break;                           // Vector  4:  ADC timing overflow
00177   case  6:                                  // Vector  6:  ADC12IFG0
00178                 adc12_result = ADC12MEM0;                       // Move results, IFG is cleared
00179                 adc12_data_ready = 1;
00180                 _BIC_SR_IRQ(LPM3_bits);                                                 // Exit active CPU
00181                 break;
00182   case  8: break;                           // Vector  8:  ADC12IFG1
00183   case 10: break;                           // Vector 10:  ADC12IFG2
00184   case 12: break;                           // Vector 12:  ADC12IFG3
00185   case 14: break;                           // Vector 14:  ADC12IFG4
00186   case 16: break;                           // Vector 16:  ADC12IFG5
