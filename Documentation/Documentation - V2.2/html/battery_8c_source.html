00102         u16 voltage;
00103         
00104         // Convert external battery voltage (ADC12INCH_11=AVCC-AVSS/2)
00105         //voltage = adc12_single_conversion(REFVSEL_2, ADC12SHT0_10, ADC12SSEL_0, ADC12SREF_1, ADC12INCH_11, ADC12_BATT_CONVERSION_TIME_USEC);
00106         voltage = adc12_single_conversion(REFVSEL_1, ADC12SHT0_10, ADC12INCH_11);
00107 
00108         // Convert ADC value to "x.xx V"
00109         // Ideally we have A11=0->AVCC=0V ... A11=4095(2^12-1)->AVCC=4V
00110         // --> (A11/4095)*4V=AVCC --> AVCC=(A11*4)/4095
00111         voltage = (voltage * 2 * 2) / 41;  
00112 
00113         // Correct measured voltage with calibration value
00114         voltage += sBatt.offset;
00115         
00116         // Discard values that are clearly outside the measurement range 
00117         if (voltage > BATTERY_HIGH_THRESHOLD) 
00118         {
00119                 voltage = sBatt.voltage;
00120         }
00121         
00122         // Filter battery voltage
00123         sBatt.voltage = ((voltage*2) + (sBatt.voltage*8))/10;
00124 
00125         // If battery voltage falls below low battery threshold, set system flag and modify LINE2 display function pointer
00126         if (sBatt.voltage < BATTERY_LOW_THRESHOLD) 
00127         {
00128                 sys.flag.low_battery = 1;
00129                 
00130                 // Set sticky battery icon
00131                 display_symbol(LCD_SYMB_BATTERY, SEG_ON);
00132         }
00133         else

00135                 sys.flag.low_battery = 0;

00138                 display_symbol(LCD_SYMB_BATTERY, SEG_OFF);
