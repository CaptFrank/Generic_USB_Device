00339 // *************************************************************************************************
00340 // @fn          simpliciti_get_ed_data_callback
00341 // @brief       Callback function to read end device data from acceleration sensor (if available) 
00342 //                              and trigger sending. Can be also be used to transmit other data at different packet rates.
00343 //                              Please observe the applicable duty limit in the chosen ISM band.
00344 // @param       none
00345 // @return      none
00346 // *************************************************************************************************
00347 void simpliciti_get_ed_data_callback(void)
00348 {
00349         static u8 packet_counter = 0;
00350 
00351         if (sRFsmpl.mode == SIMPLICITI_ACCELERATION)
00352         {
00353                 // Wait for next sample
00354                 Timer0_A4_Delay(CONV_MS_TO_TICKS(5));   
00355 
00356                 // Read from sensor if DRDY pin indicates new data (set in PORT2 ISR)
00357                 if (request.flag.acceleration_measurement && ((AS_INT_IN & AS_INT_PIN) == AS_INT_PIN))
00358                 {
00359                         // Clear flag
00360                         request.flag.acceleration_measurement = 0;
00361                         
00362                         // Get data from sensor
00363                         as_get_data(sAccel.xyz);
00364                         
00365                         // Transmit only every 3rd data set (= 33 packets / second) 
00366                         if (packet_counter++ > 1)
00367                         {
00368                                 // Reset counter
00369                                 packet_counter = 0;
00370         
00371                                 // Store XYZ data in SimpliciTI variable
00372                                 simpliciti_data[1] = sAccel.xyz[0];
00373                                 simpliciti_data[2] = sAccel.xyz[1];
00374                                 simpliciti_data[3] = sAccel.xyz[2];
00375                         
00376                                 // Trigger packet sending
00377                                 simpliciti_flag |= SIMPLICITI_TRIGGER_SEND_DATA;
00378                         }
00379                 }
00380         }
00381         else // transmit only button events
00382         {
00383                 // New button event is stored in data
00384                 if ((packet_counter == 0) && (simpliciti_data[0] & 0xF0) != 0)
00385                 {
00386                         packet_counter = 5;
00387                 }
00388                 
00389                 // Send packet several times 
00390                 if (packet_counter > 0)
00391                 {
00392                         // Clear button event when sending last packet
00393                         if (--packet_counter == 0)
00394                         {
00395                                 simpliciti_data[0] &= ~0xF0;
00396                         }
00397                         else
00398                         {
00399                                 // Trigger packet sending in regular intervals
00400                                 Timer0_A4_Delay(CONV_MS_TO_TICKS(30));
00401                                 simpliciti_flag |= SIMPLICITI_TRIGGER_SEND_DATA;
00402                         }
00403                 }
00404                 else
00405                 {
00406                         // Wait in LPM3 for next button press
00407                         _BIS_SR(LPM3_bits + GIE);       
00408                         __no_operation();
00409                 }
00410         }
00411         
00412         // Update clock every 1/1 second
00413         if (display.flag.update_time)
00414         {
00415                 display_time(LINE1, DISPLAY_LINE_UPDATE_PARTIAL);
00416                 display.flag.update_time = 0;
00417 
00418                 // Service watchdog
00419                 WDTCTL = WDTPW + WDTIS__512K + WDTSSEL__ACLK + WDTCNTCL;
00420         }
00421 }

00494 // *************************************************************************************************
00495 // @fn          simpliciti_sync_decode_ap_cmd_callback
00496 // @brief       For SYNC mode only: Decode command from access point and trigger actions. 
00497 // @param       none
00498 // @return      none
00499 // *************************************************************************************************
00500 void simpliciti_sync_decode_ap_cmd_callback(void)
00501 {
00502         u8 i;
00503         s16 t1, offset;
00504         
00505         // Default behaviour is to send no reply packets
00506         simpliciti_reply_count = 0;
00507         
00508         switch (simpliciti_data[0])
00509         {
00510                 case SYNC_AP_CMD_NOP:                   break;
00511 
00512                 case SYNC_AP_CMD_GET_STATUS:    // Send watch parameters
00513                                                                                 simpliciti_data[0]  = SYNC_ED_TYPE_STATUS;
00514                                                                                 // Send single reply packet
00515                                                                                 simpliciti_reply_count = 1;
00516                                                                                 break;
00517 
00518                 case SYNC_AP_CMD_SET_WATCH:             // Set watch parameters
00519                                                                                 sys.flag.use_metric_units = (simpliciti_data[1] >> 7) & 0x01;
00520                                                                                 sTime.hour                      = simpliciti_data[1] & 0x7F;
00521                                                                                 sTime.minute            = simpliciti_data[2];
00522                                                                                 sTime.second            = simpliciti_data[3];
00523                                                                                 sDate.year                      = (simpliciti_data[4]<<8) + simpliciti_data[5];
00524                                                                                 sDate.month             = simpliciti_data[6];
00525                                                                                 sDate.day                       = simpliciti_data[7];
00526                                                                                 sAlarm.hour                     = simpliciti_data[8];
00527                                                                                 sAlarm.minute           = simpliciti_data[9];
00528                                                                                 // Set temperature and temperature offset
00529                                                                                 t1 = (s16)((simpliciti_data[10]<<8) + simpliciti_data[11]);
00530                                                                                 offset = t1 - (sTemp.degrees - sTemp.offset);
00531                                                                                 sTemp.offset  = offset; 
00532                                                                                 sTemp.degrees = t1;                                                                     
00533                                                                                 // Set altitude
00534                                                                                 sAlt.altitude = (s16)((simpliciti_data[12]<<8) + simpliciti_data[13]);
00535                                                                                 update_pressure_table(sAlt.altitude, sAlt.pressure, sAlt.temperature);
00536                                                                                 break;
00537                                                                                                 
00538                 case SYNC_AP_CMD_GET_MEMORY_BLOCKS_MODE_1:      
00539                                                                                 // Send sequential packets out in a burst 
00540                                                                                 simpliciti_data[0]  = SYNC_ED_TYPE_MEMORY;
00541                                                                                 // Get burst start and end packet
00542                                                                                 burst_start = (simpliciti_data[1]<<8)+simpliciti_data[2];
00543                                                                                 burst_end   = (simpliciti_data[3]<<8)+simpliciti_data[4];
00544                                                                                 // Set burst mode
00545                                                                                 burst_mode = 1;
00546                                                                                 // Number of packets to send
00547                                                                                 simpliciti_reply_count = burst_end - burst_start;
00548                                                                                 break;
00549 
00550                 case SYNC_AP_CMD_GET_MEMORY_BLOCKS_MODE_2:      
00551                                                                                 // Send specified packets out in a burst 
00552                                                                                 simpliciti_data[0]  = SYNC_ED_TYPE_MEMORY;
00553                                                                                 // Store the requested packets
00554                                                                                 for (i=0; i<BM_SYNC_BURST_PACKETS_IN_DATA; i++)
00555                                                                                 {
00556                                                                                         burst_packet[i] = (simpliciti_data[i*2+1]<<8)+simpliciti_data[i*2+2];
00557                                                                                 }
00558                                                                                 // Set burst mode
00559                                                                                 burst_mode = 2;
00560                                                                                 // Number of packets to send
00561                                                                                 simpliciti_reply_count = BM_SYNC_BURST_PACKETS_IN_DATA;
00562                                                                                 break;
00563                 
00564                 case SYNC_AP_CMD_ERASE_MEMORY:  // Erase data logger memory
00565                                                                                 break;
00566                                                                                 
00567                 case SYNC_AP_CMD_EXIT:                  // Exit sync mode
00568                                                                                 simpliciti_flag |= SIMPLICITI_TRIGGER_STOP;
00569                                                                                 break;                                                                          
00570         }
00571

00590                                                                                 simpliciti_data[2]  = sTime.minute;

00593                                                                                 simpliciti_data[5]  = sDate.year & 0xFF;
00594                                                                                 simpliciti_data[6]  = sDate.month;
00595                                                                                 simpliciti_data[7]  = sDate.day;
00596                                                                                 simpliciti_data[8]  = sAlarm.hour;
00597                                                                                 simpliciti_data[9]  = sAlarm.minute;
00598                                                                                 simpliciti_data[10] = sTemp.degrees >> 8;
00599                                                                                 simpliciti_data[11] = sTemp.degrees & 0xFF;
00600                                                                                 simpliciti_data[12] = sAlt.altitude >> 8;
00601                                                                                 simpliciti_data[13] = sAlt.altitude & 0xFF;
00602                                                                                 break;
00603                                                                                 
00604                 case SYNC_ED_TYPE_MEMORY:               
00605                                                                                 if (burst_mode == 1)
