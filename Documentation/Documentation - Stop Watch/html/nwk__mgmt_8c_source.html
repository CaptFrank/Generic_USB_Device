00114 /******************************************************************************
00115  * @fn          nwk_processMgmt
00116  *
00117  * @brief       Process Management frame. Just save the frame for the Management
00118  *              app it it is a reply. If it isn't a reply, send the reply in this
00119  *              thread.
00120  *
00121  * input parameters
00122  * @param   frame   - pointer to frame to be processed
00123  *
00124  * output parameters
00125  *
00126  * @return   Keep frame for application, release frame, or replay frame.
00127  */
00128 fhStatus_t nwk_processMgmt(mrfiPacket_t *frame)
00129 {
00130   fhStatus_t   rc;
00131   uint8_t      replyType;
00132 
00133   /* If we sent this then this is the reply. Validate the
00134    * packet for reception by client app. If we didn't send
00135    * it then we are the target. send the reply.
00136    */
00137   if (SMPL_MY_REPLY == (replyType=nwk_isValidReply(frame, sTid, MB_APP_INFO_OS, MB_TID_OS)))
00138   {
00139     /* It's a match and it's a reply. Validate the received packet by
00140      * returning a 1 so it can be received by the client app.
00141      */
00142     MRFI_PostKillSem();
00143     rc = FHS_KEEP;
00144   }
00145 #if !defined( END_DEVICE )
00146   else if (SMPL_A_REPLY == replyType)
00147   {
00148     /* no match. if i'm not an ED this is a reply that should be passed on. */
00149     rc = FHS_REPLAY;
00150   }
00151 #endif  /* !END_DEVICE */
00152   else
00153   {
00154     /* no, we didn't send it. send reply if it's intended for us */
00155     if (!memcmp(MRFI_P_DST_ADDR(frame), nwk_getMyAddress(), NET_ADDR_SIZE))
00156     {
00157       smpl_send_mgmt_reply(frame);
00158 
00159       /* we're done with the frame. */
00160       rc = FHS_RELEASE;
00161     }
00162     else
00163     {
00164       rc = FHS_REPLAY;
00165     }
00166   }
00167 
00168   (void) replyType;  /* keep compiler happy */
00169 
00170   return rc;
00171 }

00216 static void send_poll_reply(mrfiPacket_t *frame)
00217 {

00219   frameInfo_t    *pOutFrame;
00220   sfClientInfo_t *pClientInfo;
00221   uint8_t         loc;
00222 
00223   /* Make sure this guy is really a client. We can tell from the source address. */
00224   if (!(pClientInfo=nwk_isSandFClient(MRFI_P_SRC_ADDR(frame), &loc)))
00225   {
00226     /* TODO: maybe send an error frame? */
00227     return;
00228   }
00229 
00230   /* If we have to resync the TID then do it based on the first
00231    * poll frame we see
00232   */
00233   if (!sSFMarker[loc])
00234   {
00235     /* If the marker flag is null then it has been initialized, i.e.,
00236      * there has been a reset. In this case infer that we need to update
00237      * a (probably) stale last TID. The test will always be true the first
00238      * time through after a client is established even when an NV restore
00239      * did not take place but this does no harm.
00240      */
00241     pClientInfo->lastTID = msgtid;
00242     sSFMarker[loc]       = 1;
00243   }
00244   /* If we've seen this poll frame before ignore it. Otherwise we
00245    * may send a stored frame when we shouldn't.
00246    */
00247   else if (nwk_checkAppMsgTID(pClientInfo->lastTID, msgtid))
00248   {
00249     pClientInfo->lastTID = msgtid;
00250   }
00251   else
00252   {
00253     return;
00254   }
00255 
00256   if (pOutFrame = nwk_getSandFFrame(frame, M_POLL_PORT_OS))
00257   {
00258     /* We need to adjust the order in the queue in this case. Currently
00259      * we know it is in the input queue and that this adjustment is safe
00260      * because we're in an ISR thread. This is a fragile fix, though, and
00261      * should be revisited when time permits.
00262      */
00263     nwk_QadjustOrder(INQ, pOutFrame->orderStamp);
00264 
00265     /* reset hop count... */
00266     PUT_INTO_FRAME(MRFI_P_PAYLOAD(&pOutFrame->mrfiPkt), F_HOP_COUNT, MAX_HOPS_FROM_AP);
00267     /* It's gonna be a forwarded frame. */
00268     PUT_INTO_FRAME(MRFI_P_PAYLOAD(&pOutFrame->mrfiPkt), F_FWD_FRAME, 0x80);
00269 
00270     nwk_sendFrame(pOutFrame, MRFI_TX_TYPE_FORCED);
00271   }
00272   else
00273   {
00274     nwk_SendEmptyPollRspFrame(frame);
00275   }
00276 
00277   return;
00278 }

00341   send.addr = (addr_t *)sAPAddr;
00342   send.msg  = msg;
00343   send.len  = sizeof(msg);
00344   send.port = SMPL_PORT_MGMT;
