00165 // @fn          init_ucs
00166 // @brief       Initialize the Unified Clock System.
00167 // @param       none
00168 // @return      none
00169 // *************************************************************************************************
00170 void init_ucs(void)
00171 {
00172         // Enable 32kHz ACLK    
00173         P5SEL |= 0x03;                            // Select XIN, XOUT on P5.0 and P5.1
00174         UCSCTL6 &= ~XT1OFF;                                       // XT1 On, Highest drive strength
00175         UCSCTL6 |= XCAP_3;                        // Internal load cap
00176 
00177         UCSCTL3 = SELA__XT1CLK;                   // Select XT1 as FLL reference
00178         UCSCTL4 = SELA__XT1CLK | SELS__DCOCLKDIV | SELM__DCOCLKDIV;      
00179         
00180         // ---------------------------------------------------------------------
00181         // Configure CPU clock for 12MHz
00182         _BIS_SR(SCG0);                  // Disable the FLL control loop
00183         UCSCTL0 = 0x0000;          // Set lowest possible DCOx, MODx
00184         UCSCTL1 = DCORSEL_5;       // Select suitable range
00185         UCSCTL2 = FLLD_1 + 0x16E;  // Set DCO Multiplier
00186         _BIC_SR(SCG0);                  // Enable the FLL control loop
00187 
00188     // Worst-case settling time for the DCO when the DCO range bits have been
00189     // changed is n x 32 x 32 x f_MCLK / f_FLL_reference. See UCS chapter in 5xx
00190     // UG for optimization.
00191     // 32 x 32 x 8 MHz / 32,768 Hz = 250000 = MCLK cycles for DCO to settle
00192     __delay_cycles(250000);
00193   
00194         // Loop until XT1 & DCO stabilizes, use do-while to insure that 
00195         // body is executed at least once
00196         do
00197         {
00198         UCSCTL7 &= ~(XT2OFFG + XT1LFOFFG + XT1HFOFFG + DCOFFG);
00199                 SFRIFG1 &= ~OFIFG;                      // Clear fault flags
00200         } while ((SFRIFG1 & OFIFG));    
00201 }
