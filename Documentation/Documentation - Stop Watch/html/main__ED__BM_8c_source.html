00069 // @fn          simpliciti_link
00070 // @brief       Init hardware and try to link to access point.
00071 // @param       none
00072 // @return      unsigned char           0 = Could not link, timeout or external cancel.
00073 //                                                                      1 = Linked successful.
00074 // *************************************************************************************************
00075 unsigned char simpliciti_link(void)
00076 {
00077   uint8_t timeout;
00078   addr_t lAddr;
00079   uint8_t i;
00080   uint8_t pwr;
00081   
00082   // Configure timer
00083   BSP_InitBoard();
00084   
00085   // Change network address to value set in calling function
00086   for (i=0; i<NET_ADDR_SIZE; i++)
00087   {
00088     lAddr.addr[i] = simpliciti_ed_address[i];
00089   }
00090   SMPL_Ioctl(IOCTL_OBJ_ADDR, IOCTL_ACT_SET, &lAddr);
00091   
00092   // Set flag   
00093   simpliciti_flag = SIMPLICITI_STATUS_LINKING;  
00094         
00095   /* Keep trying to join (a side effect of successful initialization) until
00096    * successful. Toggle LEDS to indicate that joining has not occurred.
00097    */
00098   timeout = 0;
00099   while (SMPL_SUCCESS != SMPL_Init(0))
00100   {
00101     NWK_DELAY(1000);
00102 
00103     // Service watchdog
00104         WDTCTL = WDTPW + WDTIS__512K + WDTSSEL__ACLK + WDTCNTCL;
00105     
00106     // Stop connecting after defined numbers of seconds (15)
00107     if (timeout++ > TIMEOUT) 
00108     {
00109                 // Clean up SimpliciTI stack to enable restarting
00110                 sInit_done = 0;
00111             simpliciti_flag = SIMPLICITI_STATUS_ERROR;
00112                 return (0);
00113     }
00114     
00115     // Break when flag bit SIMPLICITI_TRIGGER_STOP is set
00116     if (getFlag(simpliciti_flag, SIMPLICITI_TRIGGER_STOP)) 
00117     {
00118                 // Clean up SimpliciTI stack to enable restarting
00119         sInit_done = 0;
00120         return (0);
00121         }
00122   }
00123   
00124   // Set output power to +3.3dmB
00125   pwr = IOCTL_LEVEL_2;
00126   SMPL_Ioctl(IOCTL_OBJ_RADIO, IOCTL_ACT_RADIO_SETPWR, &pwr);
00127 
00128   /* Unconditional link to AP which is listening due to successful join. */
00129   timeout = 0;
00130   while (SMPL_SUCCESS != SMPL_Link(&sLinkID1))
00131   {
00132     NWK_DELAY(1000);
00133     
00134     // Service watchdog
00135         WDTCTL = WDTPW + WDTIS__512K + WDTSSEL__ACLK + WDTCNTCL;
00136 
00137     // Stop linking after timeout
00138     if (timeout++ > TIMEOUT) 
00139     {
00140                 // Clean up SimpliciTI stack to enable restarting
00141                 sInit_done = 0;
00142             simpliciti_flag = SIMPLICITI_STATUS_ERROR;
00143                 return (0);
00144     }
00145     
00146     // Exit when flag bit SIMPLICITI_TRIGGER_STOP is set

00150         sInit_done = 0;
