00145 /****************************************************************************************
00146  *                                 BEGIN ENDIAN SUPPORT
00147  *
00148  * Security encrypt/decrypt operates on unsigned long quantities. These must match on
00149  * source and destination platforms. These macros enforce the standard conversions.
00150  * Currently all platforms (CC2520/CC2x30 and MSP430) are little endian.
00151  *
00152  *******************   Network order for encryption is LITTLE ENDIAN   ******************
00153  *
00154  ****************************************************************************************/
00155 
00156 #if (BSP_LITTLE_ENDIAN != 0)
00157 #define   ntohs(x)    (x)
00158 #define   htons(x)    (x)
00159 
00160 #define   ntohl(x)    (x)
00161 #define   htonl(x)    (x)
00162 
00163 #else
00164 
00165 #define   ntohs(x)    (((x>>8) & 0xFF) | ((x & 0xFF)<<8))
00166 #define   htons(x)    (((x>>8) & 0xFF) | ((x & 0xFF)<<8))
00167 
00168 #define   ntohl(x)    ( ((x>>24) & 0xFF) | ((x>>8) & 0xFF00) | \
00169                         ((x & 0xFF00)<<8) | ((x & 0xFF)<<24)   \
00170                       )
00171 #define   htonl(x)    ( ((x>>24) & 0xFF) | ((x>>8) & 0xFF00) | \
00172                         ((x & 0xFF00)<<8) | ((x & 0xFF)<<24)   \
00173                       )
00174 
00175 #endif  /* (BSP_LITTLE_ENDIAN != 0) */
