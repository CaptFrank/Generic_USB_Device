00140         // Load CCR register with next capture time

00143                 // Timer interrupts occur every 32768/100 = 328 ACLK
00144                 // --> stopwatch runs too slow (1 sec nominal != 100 interupts * 328 ACLK = 32800 ACLK = 1.00098 sec)

00152                         value -= 5;

00154                         sStopwatch.swtIs10Hz = 0;       
00155                 }
00156                 else if (sStopwatch.swtIs10Hz)

00158                         value -= 3;
00159                         sStopwatch.swtIs10Hz = 0;
