00186 /***************************************************************************************
00187  *                                 ** NV Object support **
00188  *
00189  * The following object supports saving and restoring the information
00190  * necessary to save and restore a device connection context.
00191  *
00192  * On a GET interface populates the IOCTL object with the version and length (in bytes)
00193  * of the current static connection iformation area. In addition it populates the address
00194  * pointed to by 'objPtr' with the base address of the connection context. At this point
00195  * the caller can either copy to or from the address. Note that this is a dangerous
00196  * interface, as the caller is provided with direct access to the connection context.
00197  *
00198  * When restoring the connection context some sanity checks are possible. If the
00199  * version or length elements of the saved context do not match those of the current
00200  * static object the static object should not be populated. If this sanity fails the
00201  * caller is not provided with the pointer to the conneciton ocntext.
00202  *
00203  * This interface is fairly simple and it is possible to get the address of the
00204  * connection context to do a restore by simply doing a GET call. This avoids the
00205  * sanity checks. However, this is not recommended because there may be other side
00206  * effects of doing a SET that are necessary when restoring a context and are done
00207  * only when the proper option is used to restore the connection context.
00208  *
00209  *************************************************************************************/
00210 typedef struct
00211 {
00212   uint8_t    objVersion;
00213   uint16_t   objLen;
00214   uint8_t  **objPtr;
00215 } ioctlNVObj_t;

00314 /* Network applications may need to remember radio state because the user
00315  * application may choose to turn Rx off. These macros help get and restore
00316  * the radio Rx state. The macros should be in the same code block at the same level.
00317  * The argument 's' is the 'current' radio state and should be set in the code block
00318  * with a call to MRFI_GetRadioState() _before_ using the macros.
00319  *
00320  * Used extensively by NWK but user applications may use them as well. But it is
00321  * much more liekly that an application will know the radio state since it likely
00322  * will have set it with IOCTL calls. Requires mrfi.h.
00323  */
00324 #define NWK_CHECK_FOR_SETRX(s)  if (MRFI_RADIO_STATE_RX != s)    \
00325                                 {                                \
00326                                   if (MRFI_RADIO_STATE_OFF == s) \
00327                                   {                              \
00328                                     MRFI_WakeUp();               \
00329                                   }                              \
00330                                   MRFI_RxOn();                   \
00331                                 }
00332 
00333 #define NWK_CHECK_FOR_RESTORE_STATE(s) if (MRFI_RADIO_STATE_RX != s)    \
00334                                        {                                \
00335                                          if (MRFI_RADIO_STATE_OFF == s) \
00336                                          {                              \
00337                                            MRFI_Sleep();                \
00338                                          }                              \
00339                                          else                           \
00340                                          {                              \
00341                                            MRFI_RxIdle();               \
00342                                          }                              \
00343                                        }
00344 #endif
