00074 // @fn          display_charSegments
00075 // @brief       Write to 7-segment characters.
00076 // @param       u8 segment                  A valid LCD segment 
00077 //                              u8 charSegments         Segments of a character to display (SEG_A .. SEG_F)
00078 //                              u8 mode                 SEG_ON, SEG_OFF, SEG_BLINK
00079 // @return      none
00080 // *************************************************************************************************
00081 void display_charSegments(u8 segment, u8 charSegments, u8 mode)
00082 {
00083         u8 * lcdmem;                    // Pointer to LCD memory
00084         u8 bitmask;                     // Bitmask for character
00085         u8 bits, bits1;         // Bits to write
00086         
00087         // Write to single 7-segment character
00088         if ((segment >= LCD_SEG_L1_3) && (segment <= LCD_SEG_L2_DP))
00089         {
00090                 // Get LCD memory address for segment from table
00091                 lcdmem = (u8 *)segments_lcdmem[segment];
00092 
00093                 // Get bitmask for character from table
00094                 bitmask = segments_bitmask[segment];
00095                 
00096                 // Segments of a char to display
00097                 bits = charSegments;
00098 
00099                 // When addressing LINE2 7-segment characters need to swap high- and low-nibble,
00100                 // because LCD COM/SEG assignment is mirrored against LINE1
00101                 if (segment >= LCD_SEG_L2_5) 
00102                 {
00103                         bits1 = ((bits << 4) & 0xF0) | ((bits >> 4) & 0x0F);
00104                         bits = bits1;
00105 
00106                         // When addressing LCD_SEG_L2_5, need to convert ASCII '1' and 'L' to 1 bit,
00107                         // because LCD COM/SEG assignment is special for this incomplete character
00108                         if (segment == LCD_SEG_L2_5) 
00109                         {
00110                                 if (charSegments == (SEG_B+SEG_C)) bits = BIT7;
00111                                 else  bits = 0;
00112                         }
00113                 }
00114                 
00115                 // Physically write to LCD memory               
00116                 write_lcd_mem(lcdmem, bits, bitmask, mode);
00117         }
00118 }
