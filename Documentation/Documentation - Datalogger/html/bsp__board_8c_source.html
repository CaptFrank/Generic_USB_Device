00064 /**************************************************************************************************
00065  * @fn          SetVCore
00066  *
00067  * @brief       SetVCore level is called from the user API. Change level by one step only.
00068  *
00069  * @param       level - VcCore level
00070  *
00071  * @return      none
00072  **************************************************************************************************
00073  */
00074 static void SetVCore (uint8_t level)
00075 {
00076   uint8_t actLevel;
00077   do {
00078     actLevel = PMMCTL0_L & PMMCOREV_3;
00079     if (actLevel < level)
00080       SetVCoreUp(++actLevel);               /* Set VCore (step by step) */
00081     if (actLevel > level)
00082       SetVCoreDown(--actLevel);             /* Set VCore (step by step) */
00083   }while (actLevel != level);
00084 }

00086 /**************************************************************************************************
00087  * @fn          SetVCoreUp
00088  *
00089  * @brief       Set VCore up. Change level by one step only.
00090  *
00091  * @param       level - VcCore level
00092  *
00093  * @return      none
00094  *************************************************************************************************
00095  */
00096 static void SetVCoreUp (uint8_t level)
00097 {
00098   PMMCTL0_H = 0xA5;                          /* Open PMM module registers for write access */
00099 
00100   SVSMHCTL = SVSHE + SVSHRVL0 * level + SVMHE + SVSMHRRL0 * level; /* Set SVS/M high side to new level */
00101 
00102   SVSMLCTL = SVSLE + SVMLE + SVSMLRRL0 * level; /* Set SVM new Level */
00103   while ((PMMIFG & SVSMLDLYIFG) == 0);       /* Wait till SVM is settled (Delay) */
00104   PMMCTL0_L = PMMCOREV0 * level;             /* Set VCore to x */
00105   PMMIFG &= ~(SVMLVLRIFG + SVMLIFG);         /* Clear already set flags */
00106   if ((PMMIFG & SVMLIFG))
00107     while ((PMMIFG & SVMLVLRIFG) == 0);      /* Wait till level is reached */
00108 
00109   SVSMLCTL = SVSLE + SVSLRVL0 * level + SVMLE + SVSMLRRL0 * level;  /* Set SVS/M Low side to new level */
00110   PMMCTL0_H = 0x00;                          /* Lock PMM module registers for write access */
00111 }

00124 {

00147 /**************************************************************************************************
00148  * @fn          Bsp_SetClocks
00149  *
00150  * @brief       Set up system clocks.
00151  *
00152  * @param       none
00153  *
00154  * @return      none
00155  **************************************************************************************************
00156  */
00157 static void Bsp_SetClocks(void)
00158 {
00159   /* Configure CPU clock for 12MHz */
00160 
00161   /* If clock settings are changed, remember to update BSP_TIMER_CLK_MHZ.
00162    * Otherwise, all timer settings would be incorrect.
00163    */
00164   UCSCTL3 |= SELREF_2;                      /* Set DCO FLL reference = REFO */
00165   UCSCTL4 |= SELA_2;                        /* Set ACLK = REFO */
00166 
00167   __bis_SR_register(SCG0);                  /* Disable the FLL control loop */
00168   UCSCTL0 = 0x0000;                         /* Set lowest possible DCOx, MODx */
00169   UCSCTL1 = DCORSEL_5;                      /* Select DCO range 24MHz operation */
00170   UCSCTL2 = FLLD_1 + 374;                   /* Set DCO Multiplier for 12MHz */
00171                                             /* (N + 1) * FLLRef = Fdco */
00172                                             /* (374 + 1) * 32768 = 12MHz */
00173                                             /* Set FLL Div = fDCOCLK/2 */
00174   __bic_SR_register(SCG0);                  /* Enable the FLL control loop */
00175 
00176   /* Worst-case settling time for the DCO when the DCO range bits have been
00177    * changed is n x 32 x 32 x f_MCLK / f_FLL_reference.
00178    * 32 x 32 x 12 MHz / 32,768 Hz = 375000 = MCLK cycles for DCO to settle
00179    */
00180   __delay_cycles(375000);
00181         
00182   /* Loop until XT1,XT2 & DCO fault flag is cleared */
00183   do
00184   {
00185     UCSCTL7 &= ~(XT2OFFG + XT1LFOFFG + DCOFFG);
00186                                             /* Clear XT2,XT1,DCO fault flags */
00187     SFRIFG1 &= ~OFIFG;                      /* Clear fault flags */
00188   }while (SFRIFG1&OFIFG);                   /* Test oscillator fault flag */
00189 
00190   /* Select REFO as ACLK source and DCOCLK as MCLK and SMCLK source */
00191   UCSCTL4 = SELA__REFOCLK | SELS__DCOCLKDIV | SELM__DCOCLKDIV;
00192 }
