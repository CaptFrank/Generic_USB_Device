00061 void sx_rfbsl(u8 line)
00062 {
00063         // Exit if battery voltage is too low for radio operation
00064         if (sys.flag.low_battery) return;
00065         
00066         // Exit if BlueRobin stack is active
00067         if (is_bluerobin()) return;
00068         
00069         // Exit if SimpliciTI stack is active
00070         if (is_rf()) return;
00071         
00072         // Before entering RFBSL clear the LINE1 Symbols
00073         display_symbol(LCD_SYMB_AM, SEG_OFF);
00074 
00075         clear_line(LINE1);
00076         
00077         // Write RAM to indicate we will be downloading the RAM Updater first
00078         display_chars(LCD_SEG_L1_3_0, (u8 *)" RAM", SEG_ON);
00079         
00080         // Call RFBSL
00081         CALL_RFSBL();
00082 }
