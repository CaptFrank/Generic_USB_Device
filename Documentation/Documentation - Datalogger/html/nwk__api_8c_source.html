00060 /* These defines are in support an application listening for a link frame to
00061  * terminate after some amount of time. The intention is that this guard be
00062  * the exception. The intention of the SimpliciTI design is that the
00063  * temporal contiguity between the listen and the reception of the link frame
00064  * from the peer be very tight. The SMPL_LinkListen() should be termninated
00065  * by the reception of the link frame. But in case it does not receive the frame
00066  * the support below allows intervention by the application.
00067  */

00136 #if defined( FREQUENCY_AGILITY )

00164 /******************************************************************************
00165  * @fn          SMPL_LinkListen
00166  *
00167  * @brief       Listen for a link frame from a 'client' device.
00168  *
00169  * input parameters
00170  *
00171  * output parameters
00172  * @param   linkID     - pointer to Link ID to be used by application to
00173  *                       read and write to the linked peer.
00174  *
00175  * @return   status of operation.
00176  *             SMPL_SUCCESS
00177  *             SMPL_TIMEOUT  No link frame received during listen interval.
00178 *                            Interval set in #defines above. linkID not valid.
00179  *
00180  */
00181 
00182 smplStatus_t SMPL_LinkListen(linkID_t *linkID)
00183 {
00184   uint8_t  radioState = MRFI_GetRadioState();
00185   uint16_t i;
00186   linkID_t locLinkID;
00187 
00188   /* Set the context. We want to reject any link frames received if
00189    * we're not listening. For example if we're an AP we are in
00190    * promiscuous mode and we'll see any broadcast link frames.
00191    */
00192   nwk_setListenContext(LINK_LISTEN_ON);
00193 
00194   NWK_CHECK_FOR_SETRX(radioState);
00195 
00196   for (i=0; i<LINKLISTEN_POLL_COUNT; ++i)
00197   {
00198     /* check the semaphore. local port is assigned when the reply is sent. */
00199     if ((locLinkID=nwk_getLocalLinkID()))
00200     {
00201       break;
00202     }
00203     NWK_DELAY(LINKLISTEN_POLL_PERIOD_MS);
00204   }
00205 
00206   NWK_CHECK_FOR_RESTORE_STATE(radioState);
00207 
00208   /* If the listen is terminated without hearing a message and setting a
00209    * link ID the listen context must be explicitly turned off.
00210    */
00211   if (!(locLinkID))
00212   {
00213     nwk_setListenContext(LINK_LISTEN_OFF);
00214     return SMPL_TIMEOUT;
00215   }
00216 
00217   *linkID = locLinkID;
00218 
00219   return SMPL_SUCCESS;
00220 }

00283 #endif
00284 
00285   /* we have the connection info for this Link ID. make sure it is valid. */
00286    if (!pCInfo || ((rc=nwk_checkConnInfo(pCInfo, CHK_TX)) != SMPL_SUCCESS))
00287   {
00288     return rc;
00289   }
00290 
00291   /* parameter sanity check... */
00292   if (!msg || (len > MAX_APP_PAYLOAD))
00293   {
00294     return rc;
00295   }
00296 
00297   /* Build an outgoing message frame destined for the port from the
00298    * connection info using the destination address also from the
00299    * connection info.
00300    */
00301   if (SMPL_TXOPTION_NONE == options)

00303     pFrameInfo = nwk_buildFrame(pCInfo->portTx, msg, len, pCInfo->hops2target);

00320   else
00321   {
00322     return SMPL_BAD_PARAM;
00323   }
00324 
00325   if (!pFrameInfo)
00326   {
00327     return SMPL_NOMEM;
00328   }

00330 
00331 #if defined(SMPL_SECURE)

00342 
00343 #if defined(ACCESS_POINT)

00351   }
00352   else
00353 #endif  /* ACCESS_POINT */
00354   {
00355     rc = nwk_sendFrame(pFrameInfo, MRFI_TX_TYPE_CCA);
00356   }
00357 
00358 #if !defined(APP_AUTO_ACK)
00359   /* save a little code space with this #if */
00360   (void) ackreq;    /* keep compiler happy */
00361   return rc;
00362 #else

00364   if (SMPL_SUCCESS != rc || !ackreq)
00365   {
00366     return rc;
00367   }

00369   NWK_CHECK_FOR_SETRX(radioState);

00375 
00376     /* If the saved TID hasn't been reset then we never got the ack. */
00377     BSP_ENTER_CRITICAL_SECTION(intState);
00378     if (pCInfo->ackTID)
00379     {
00380       pCInfo->ackTID = 0;
00381       rc = SMPL_NO_ACK;
00382     }
00383     BSP_EXIT_CRITICAL_SECTION(intState);
00384   }

00387 #endif  /* APP_AUTO_ACK */

00390 /**************************************************************************************
00391  * @fn          SMPL_Receive
00392  *
00393  * @brief       Receive a message from a peer application.
00394  *
00395  * input parameters
00396  * @param   lid     - Link ID (port) from application
00397  *
00398  *
00399  * output parameters
00400  * @param   msg     - pointer to where received message should be copied.
00401  *                    buffer should be of size == MAX_APP_PAYLOAD
00402  * @param   len     - pointer to receive length of received message
00403  *
00404  * @return    Status of operation.
00405  *            Caller should not use the value returned in 'len' to decide
00406  *            whether there is a frame or not. It could be useful to the
00407  *            Caller to distinguish between no frame and a frame with no data.
00408  *            For example, in the polling case a frame with no application payload
00409  *            is the way the AP conveys that there are no frames waiting.
00410  *
00411  *              SMPL_SUCCESS
00412  *
00413  *              SMPL_BAD_PARAM  No valid Connection Table entry for Link ID
00414  *                              Data in Connection Table entry bad
00415  *              SMPL_NO_FRAME   No frame received.
00416  *              SMPL_NO_PAYLOAD Frame received with no payload (not necessarily
00417  *                              an error and could be deduced by application
00418  *                              because the returned length will be 0)
00419  *
00420  *            Polling device only:
00421  *
00422  *              SMPL_TIMEOUT        No response from Access Point
00423  *              SMPL_NO_AP_ADDRESS  Access Point address unknown
00424  *              SMPL_TX_CCA_FAIL    Could not send poll frame
00425  *              SMPL_NOMEM          No memory in output frame queue
00426  *              SMPL_NO_CHANNEL     Frequency Agility enabled and could not find channel
00427  */
00428 smplStatus_t SMPL_Receive(linkID_t lid, uint8_t *msg, uint8_t *len)
00429 {
00430   connInfo_t  *pCInfo = nwk_getConnInfo(lid);
00431   smplStatus_t rc = SMPL_BAD_PARAM;
00432   rcvContext_t rcv;
00433 
00434   if (!pCInfo || ((rc=nwk_checkConnInfo(pCInfo, CHK_RX)) != SMPL_SUCCESS))
00435   {
00436     return rc;
00437   }
00438 
00439   rcv.type  = RCV_APP_LID;
00440   rcv.t.lid = lid;
00441 
00442 #if defined(RX_POLLS)
00443   {
00444     uint8_t numChans  = 1;
00445 #if defined(FREQUENCY_AGILITY)
00446     freqEntry_t chans[NWK_FREQ_TBL_SIZE];
00447     uint8_t     scannedB4 = 0;
00448 #endif
00449 
00450     do
00451     {
00452       uint8_t radioState = MRFI_GetRadioState();
00453 
00454       /* I'm polling. Do the poll to stimulate the sending of a frame. If the
00455        * frame has application length of 0 it means there were no frames.  If
00456        * no reply is received infer that the channel is changed. We then need
00457        * to scan and then retry the poll on each channel returned.
00458        */
00459       if (SMPL_SUCCESS != (rc=nwk_poll(pCInfo->portRx, pCInfo->peerAddr)))
00460       {
00461         /* for some reason couldn't send the poll out. */
00462         return rc;
00463       }
00464 
00465       /* do this before code block below which may reset it. */
00466       numChans--;
00467 
00468       /* Wait until there's a frame. if the len is 0 then return SMPL_NO_FRAME
00469        * to the caller. In the poll case the AP always sends something.
00470        */
00471       NWK_CHECK_FOR_SETRX(radioState);
00472       NWK_REPLY_DELAY();
00473       NWK_CHECK_FOR_RESTORE_STATE(radioState);
00474 
00475       /* TODO: deal with pending */
00476       rc = nwk_retrieveFrame(&rcv, msg, len, 0, 0);
00477 
00478 #if defined(FREQUENCY_AGILITY)
00479       if (SMPL_SUCCESS == rc)
00480       {
00481         /* we received something... */
00482         return (*len) ? SMPL_SUCCESS : SMPL_NO_PAYLOAD;
00483       }
00484 
00485       /* No reply. scan for other channel(s) if we haven't already. Then set
00486        * one and try again.
00487        */
00488       if (!scannedB4)
00489       {
00490         numChans  = nwk_scanForChannels(chans);
00491         scannedB4 = 1;
00492       }
00493       if (numChans)
00494       {
00495         nwk_setChannel(&chans[numChans-1]);
00496       }
00497 #else /*  FREQUENCY_AGILITY */
00498       return (*len) ? rc : ((SMPL_SUCCESS == rc) ? SMPL_NO_PAYLOAD : SMPL_TIMEOUT);
00499 #endif
00500     } while (numChans);
00501   }
00502 
00503 #if defined(FREQUENCY_AGILITY)
00504   return SMPL_NO_CHANNEL;
00505 #endif
00506 
00507 #else  /* RX_POLLS */
00508   return nwk_retrieveFrame(&rcv, msg, len, 0, 0);
00509 #endif  /* RX_POLLS */
00510 }
