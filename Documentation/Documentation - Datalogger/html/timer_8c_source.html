00095 // @brief       Set Timer0 to a period of 1 or 2 sec. IRQ TACCR0 is asserted when timer overflows.
00096 // @param       none
00097 // @return      none

00099 void Timer0_Init(void)
00100 {
00101         // Set interrupt frequency to 1Hz
00102         TA0CCR0   = 32768 - 1;                
00103 
00104         // Enable timer interrupt
