00121 // @return      none

00123 void agility_tick(void)
00124 {
00125   // Make sure it is something running before we do anything else
00126   if (sAgil.state == AGIL_RUN)
00127   {
00128       u8 i;
00129 
00130       for(i = 0; i < 3; i++)
00131       {
00132           // Get data from sensor
00133           sAgil.xyz[i] = sAccel.xyz[i];  
00134           // Save previous data
00135           sAgil.PreviousData[i] = sAgil.CurrentData[i];
00136           // Calculate acceleration (result without sign information)
00137           sAgil.CurrentData[i] = convert_acceleration_value_to_mgrav(sAgil.xyz[i]) / 10;
00138           // If acceleraition is negative
00139           if (acceleration_value_is_positive(sAgil.xyz[i]) == 0) { sAgil.CurrentData[i] *= (-1); }
00140           // Accumulate (max. increment per sec is about 3 * 2000 / 10 / 10 = 60)
00141           sAgil.AgilityIndicator += ( abs(sAgil.PreviousData[i] - sAgil.CurrentData[i]) / 10);
00142       }
00143       
00144       // Decrease timeout counter
00145       if(sAgil.timeout > 1) 
00146       {
00147           sAgil.timeout--;
00148           // Check timeout of module acceleration.c
00149           if(sAccel.timeout < ACCEL_TIMEOUT_PREVENTION)
00150               { 
00151               // Set acceleration timeout counter to prevent an undesired stop
00152               sAccel.timeout = ACCEL_TIMEOUT_PREVENTION;
00153               }
00154       }
00155       // Timeout occured; Save current measurement to history; Stop measurement, if number of programmed cycles elapsed.
00156       else // sAgil.timeout == 1
00157       { 
00158           if(sAgil.cycles > 1)
00159           {
00160               sAgil.timeout = sAgil.TimeoutSetting;
00161               sAgil.cycles--;
00162           }
00163           else // sAgil.cycles == 1
00164           {
00165               sAgil.cycles = 0;
00166               stop_agility();
00167           }

00169       sAgil.AgilityIndicator = 0;
