00233 // @fn          itoa
00234 // @brief       Generic integer to array routine. Converts integer n to string.
00235 //                              Default conversion result has leading zeros, e.g. "00123"
00236 //                              Option to convert leading '0' into whitespace (blanks)
00237 // @param       u32 n                   integer to convert
00238 //                              u8 digits               number of digits
00239 //                              u8 blanks               fill up result string with number of whitespaces instead of leading zeros  
00240 // @return      u8                              string
00241 // *************************************************************************************************
00242 u8 * itoa(u32 n, u8 digits, u8 blanks)
00243 {
00244         u8 i;
00245         u8 digits1 = digits;
00246         
00247         // Preset result string
00248         memcpy(itoa_str, "0000000", 7);
00249 
00250         // Return empty string if number of digits is invalid (valid range for digits: 1-7)
00251         if ((digits == 0) || (digits > 7)) return (itoa_str);
00252         
00253         // Numbers 0 .. 180 can be copied from itoa_conversion_table without conversion
00254         if (n <= 180)
00255         {
00256                 if (digits >= 3)
00257                 {
00258                         memcpy(itoa_str+(digits-3), itoa_conversion_table[n], 3);
00259                 }
00260                 else // digits == 1 || 2  
00261                 {
00262                         memcpy(itoa_str, itoa_conversion_table[n]+(3-digits), digits);
00263                 }
00264         }
00265         else // For n > 180 need to calculate string content
00266         {
00267                 // Calculate digits from least to most significant number
00268                 do                                                              
00269                 {     
00270                         itoa_str[digits-1] = n % 10 + '0';      
00271                         n /= 10;
00272                 } while (--digits > 0);                 
00273         }
00274 
00275         // Remove specified number of leading '0', always keep last one
00276         i = 0;  
00277         while ((itoa_str[i] == '0') && (i < digits1-1)) 
00278         {
00279                 if (blanks > 0)
00280                 {
00281                         // Convert only specified number of leading '0'
00282                         itoa_str[i]=' ';
00283                         blanks--;
00284                 }
00285                 i++;
00286         }
00287         
00288         return (itoa_str);

00397 // @fn          display_char
00398 // @brief       Write to 7-segment characters.
00399 // @param       u8 segment              A valid LCD segment 
00400 //                              u8 chr                  Character to display
00401 //                              u8 mode         SEG_ON, SEG_OFF, SEG_BLINK
00402 // @return      none

00405 {

00407         u8 bitmask;                     // Bitmask for character
00408         u8 bits, bits1;         // Bits to write
00409         
00410         // Write to single 7-segment character
00411         if ((segment >= LCD_SEG_L1_3) && (segment <= LCD_SEG_L2_DP))
00412         {
00413                 // Get LCD memory address for segment from table
00414                 lcdmem = (u8 *)segments_lcdmem[segment];
00415 
00416                 // Get bitmask for character from table
00417                 bitmask = segments_bitmask[segment];
00418                 
00419                 // Get bits from font set
00420                 if ((chr >= 0x30) && (chr <= 0x5A)) 
00421                 {
00422                         // Use font set
00423                         bits = lcd_font[chr-0x30];
00424                 }
00425                 else if (chr == 0x2D)
00426                 {
00427                         // '-' not in font set
00428                         bits = BIT1;
00429                 }
00430                 else
00431                 {
00432                         // Other characters map to ' ' (blank)
00433                         bits = 0;
00434                 }
00435 
00436                 // When addressing LINE2 7-segment characters need to swap high- and low-nibble,
00437                 // because LCD COM/SEG assignment is mirrored against LINE1
00438                 if (segment >= LCD_SEG_L2_5) 
00439                 {
00440                         bits1 = ((bits << 4) & 0xF0) | ((bits >> 4) & 0x0F);
00441                         bits = bits1;
00442 
00443                         // When addressing LCD_SEG_L2_5, need to convert ASCII '1' and 'L' to 1 bit,
00444                         // because LCD COM/SEG assignment is special for this incomplete character
00445                         if (segment == LCD_SEG_L2_5) 
00446                         {
00447                                 if ((chr == '1') || (chr == 'L')) bits = BIT7;
00448                         }
00449                 }
00450                 
00451                 // Physically write to LCD memory               
00452                 write_lcd_mem(lcdmem, bits, bitmask, mode);
00453         }
